# CLI Tool Example

Build a robust CLI tool using better-py.

## Code

```python
#!/usr/bin/env python
import argparse
from better_py import Result

def parse_args(args: list[str]) -> Result[argparse.Namespace, str]:
    """Parse CLI arguments"""
    parser = argparse.ArgumentParser(description="My CLI Tool")
    parser.add_argument("--input", required=True)
    parser.add_argument("--output", required=True)
    
    try:
        return Result.ok(parser.parse_args(args))
    except Exception as e:
        return Result.error(str(e))

def main():
    args = parse_args(sys.argv[1:])
    
    if args.is_ok():
        processed = process(args.unwrap())
        print(f"Success: {processed}")
    else:
        print(f"Error: {args.unwrap_error()}")

if __name__ == "__main__":
    main()
```
